Python Dates

Python has a built-in module called datetime to work with dates and times.

Import the datetime Module

import datetime

Get Current Date and Time

import datetime

x = datetime.datetime.now()
print(x)  # 2023-12-07 14:30:25.123456

Date Object

import datetime

# Create a date object
x = datetime.datetime(2023, 12, 7)
print(x)  # 2023-12-07 00:00:00

strftime() Method

The strftime() method formats date objects into readable strings.

import datetime

x = datetime.datetime.now()

# Format codes
print(x.strftime("%Y"))     # 2023 (4-digit year)
print(x.strftime("%y"))     # 23 (2-digit year)
print(x.strftime("%m"))     # 12 (month as number)
print(x.strftime("%B"))     # December (month name)
print(x.strftime("%b"))     # Dec (month abbreviation)
print(x.strftime("%d"))     # 07 (day of month)
print(x.strftime("%A"))     # Thursday (weekday name)
print(x.strftime("%a"))     # Thu (weekday abbreviation)
print(x.strftime("%H"))     # 14 (hour 24-hour format)
print(x.strftime("%I"))     # 02 (hour 12-hour format)
print(x.strftime("%M"))     # 30 (minute)
print(x.strftime("%S"))     # 25 (second)
print(x.strftime("%p"))     # PM (AM/PM)

Common Format Codes

Code Description Example
%Y 4-digit year 2023
%y 2-digit year 23
%m Month as number 12
%B Full month name December
%b Month abbreviation Dec
%d Day of month 07
%A Full weekday name Thursday
%a Weekday abbreviation Thu

Date Arithmetic

import datetime

# Current date
today = datetime.date.today()
print(f"Today: {today}")

# Add days
future_date = today + datetime.timedelta(days=30)
print(f"30 days from now: {future_date}")

# Subtract days
past_date = today - datetime.timedelta(days=30)
print(f"30 days ago: {past_date}")

# Add weeks
future_week = today + datetime.timedelta(weeks=2)
print(f"2 weeks from now: {future_week}")

Compare Dates

import datetime

date1 = datetime.date(2023, 12, 7)
date2 = datetime.date(2023, 12, 10)

if date1 < date2:
    print("date1 is earlier than date2")
elif date1 > date2:
    print("date1 is later than date2")
else:
    print("date1 and date2 are the same")

Time Objects

import datetime

# Create a time object
time_obj = datetime.time(14, 30, 25)
print(time_obj)  # 14:30:25

# Access time components
print(time_obj.hour)    # 14
print(time_obj.minute)  # 30
print(time_obj.second)  # 25

Timezone

import datetime
import pytz

# Get current time in different timezones
utc_now = datetime.datetime.now(pytz.UTC)
print(f"UTC: {utc_now}")

# Convert to different timezone
ny_tz = pytz.timezone('America/New_York')
ny_time = utc_now.astimezone(ny_tz)
print(f"New York: {ny_time}")

Parse Date from String

from datetime import datetime

# Parse date from string
date_string = "2023-12-07 14:30:25"
parsed_date = datetime.strptime(date_string, "%Y-%m-%d %H:%M:%S")
print(parsed_date)  # 2023-12-07 14:30:25
