Python User Input

Python allows you to get user input using the input() function.

Basic Input

# Get user input
name = input("Enter your name: ")
print(f"Hello, {name}!")

Input with Different Data Types

# String input
name = input("Enter your name: ")

# Integer input
age = int(input("Enter your age: "))

# Float input
height = float(input("Enter your height: "))

print(f"Name: {name}, Age: {age}, Height: {height}")

Input Validation

# Validate integer input
while True:
    try:
        age = int(input("Enter your age: "))
        break
    except ValueError:
        print("Please enter a valid number!")

print(f"Your age is: {age}")

Multiple Inputs

# Get multiple inputs
name = input("Enter your name: ")
age = int(input("Enter your age: "))
city = input("Enter your city: ")

print(f"Hello {name}, you are {age} years old and live in {city}")

Input with Default Values

# Input with default value
name = input("Enter your name (or press Enter for 'Guest'): ")
if not name:
    name = "Guest"

print(f"Hello, {name}!")

Password Input

import getpass

# Hidden password input
password = getpass.getpass("Enter your password: ")
print("Password entered successfully!")

Menu Selection

# Menu with user input
print("1. Add")
print("2. Subtract")
print("3. Multiply")
print("4. Divide")

choice = input("Enter your choice (1-4): ")

if choice == "1":
    print("You selected Add")
elif choice == "2":
    print("You selected Subtract")
elif choice == "3":
    print("You selected Multiply")
elif choice == "4":
    print("You selected Divide")
else:
    print("Invalid choice!")

Input in Loops

# Get input in a loop
numbers = []
for i in range(3):
    num = int(input(f"Enter number {i+1}: "))
    numbers.append(num)

print(f"Numbers entered: {numbers}")
print(f"Sum: {sum(numbers)}")

File Input

# Read from file
filename = input("Enter filename: ")
try:
    with open(filename, 'r') as file:
        content = file.read()
        print(content)
except FileNotFoundError:
    print("File not found!")

Input with Confirmation

# Input with confirmation
while True:
    name = input("Enter your name: ")
    confirm = input(f"Is '{name}' correct? (y/n): ")
    
    if confirm.lower() == 'y':
        break
    elif confirm.lower() == 'n':
        continue
    else:
        print("Please enter 'y' or 'n'")

print(f"Confirmed name: {name}")
