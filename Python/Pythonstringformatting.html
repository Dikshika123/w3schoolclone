Python String Formatting

String formatting allows you to format strings in various ways to make them more readable and dynamic.

String format() Method

# Basic formatting
age = 25
name = "John"
text = "My name is {} and I am {} years old"
print(text.format(name, age))  # My name is John and I am 25 years old

Indexed Formatting

# Using index numbers
text = "My name is {0} and I am {1} years old. {0} is a great name!"
print(text.format("John", 25))
# My name is John and I am 25 years old. John is a great name!

Named Formatting

# Using named placeholders
text = "My name is {name} and I am {age} years old"
print(text.format(name="John", age=25))
# My name is John and I am 25 years old

f-strings (Formatted String Literals)

# f-strings (Python 3.6+)
name = "John"
age = 25
text = f"My name is {name} and I am {age} years old"
print(text)  # My name is John and I am 25 years old

f-strings with Expressions

# f-strings with expressions
price = 19.99
quantity = 3
total = f"Total: ${price * quantity:.2f}"
print(total)  # Total: $59.97

Number Formatting

# Number formatting
number = 3.14159

# Decimal places
print(f"{number:.2f}")    # 3.14
print(f"{number:.4f}")    # 3.1416

# Width and alignment
print(f"{number:10.2f}")  #       3.14
print(f"{number:<10.2f}") # 3.14      
print(f"{number:>10.2f}") #       3.14
print(f"{number:^10.2f}") #   3.14

String Formatting with %

# Old style formatting with %
name = "John"
age = 25
print("My name is %s and I am %d years old" % (name, age))
# My name is John and I am 25 years old

# Different format specifiers
print("Price: $%.2f" % 19.99)      # Price: $19.99
print("Percentage: %.1f%%" % 85.5)  # Percentage: 85.5%

Format Specifiers

Specifier Description Example
s String %s
d Integer %d
f Float %f
e Scientific notation %e
g General format %g

Alignment and Padding

# Alignment
text = "Hello"
print(f"{text:<10}")  # Hello     
print(f"{text:>10}")  #      Hello
print(f"{text:^10}")  #   Hello   

# Padding with characters
print(f"{text:*<10}")  # Hello*****
print(f"{text:*>10}")  # *****Hello
print(f"{text:*^10}")  # **Hello***

Date and Time Formatting

from datetime import datetime

now = datetime.now()
print(f"Current date: {now:%Y-%m-%d}")        # 2023-12-07
print(f"Current time: {now:%H:%M:%S}")        # 14:30:25
print(f"Full datetime: {now:%Y-%m-%d %H:%M:%S}")  # 2023-12-07 14:30:25

Template Strings

from string import Template

# Template strings
template = Template("Hello $name, you have $count messages")
result = template.substitute(name="John", count=5)
print(result)  # Hello John, you have 5 messages

String Methods for Formatting

text = "hello world"

# Case methods
print(text.upper())      # HELLO WORLD
print(text.lower())      # hello world
print(text.title())      # Hello World
print(text.capitalize()) # Hello world

# Alignment methods
print(text.ljust(20))    # hello world         
print(text.rjust(20))    #          hello world
print(text.center(20))   #     hello world
